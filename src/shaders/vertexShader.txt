#version 400 core

in vec3 position;

out vec3 color;

void main( void ){
	// first figure out where to put the pixel
	gl_Position = vec4(position.x,position.y,position.z,1.0);
	// now set the output with a simple formula that 
	// makes Red and Blue change based on X and Y
	// Gree remains
	color = vec3(position.x+0.5,1.0,position.y+0.5);

}
